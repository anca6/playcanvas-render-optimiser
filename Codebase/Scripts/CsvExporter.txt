import { Script } from 'playcanvas';

export class CsvExporter extends Script {
    static scriptName = 'csvExporter';

    /**
     * @attribute @type {boolean}
     */
    logWhenDone = true;

    /**
     * @attribute @type {boolean}
     */
    downloadWhenDone = true;

    /**
     * @attribute @type {string}
     */
    fileName = 'baseline_results.csv';

    initialize() {
        this.headers = [];
        this.rows = [];
    }

    // Starts a new CSV export.
    startNew(headers) {
        this.headers = headers || [];
        this.rows = [];
    }

    addRow(row) {
        if (!this.headers.length) {
            this.headers = Object.keys(row);
        }

        this.rows.push(row);

        console.log('[CSV EXPORTER] Added row:', row);
    }

    // Builds the CSV text.
    buildCsv() {
        const lines = [];

        // adding the header row first.
        lines.push(this.headers.map(value => this.escapeCsv(value)).join(','));

        // adding every data row under the headers.
        for (const row of this.rows) {
            const line = this.headers
                .map(header => this.escapeCsv(row[header]))
                .join(',');

            lines.push(line);
        }

        return lines.join('\n');
    }

    // Finishes the CSV export.
    finish() {
        const csv = this.buildCsv();

        if (this.logWhenDone) {
            console.log('\n===== BENCHMARK CSV START =====\n');
            console.log(csv);
            console.log('\n===== BENCHMARK CSV END =====\n');
        }

        if (this.downloadWhenDone) {
            this.downloadCsv(csv, this.fileName);
        }

        return csv;
    }

    escapeCsv(value) {
        if (value === undefined || value === null) {
            return '';
        }

        const text = String(value);

        if (text.includes(',') || text.includes('"') || text.includes('\n')) {
            return `"${text.replace(/"/g, '""')}"`;
        }

        return text;
    }

    // Downloads the CSV file.
    downloadCsv(csv, fileName) {
        const blob = new Blob([csv], { type: 'text/csv;charset=utf-8;' });
        const url = URL.createObjectURL(blob);

        const link = document.createElement('a');
        link.href = url;
        link.download = fileName;
        link.style.display = 'none';

        document.body.appendChild(link);
        link.click();
        document.body.removeChild(link);

        URL.revokeObjectURL(url);
    }
}
