// Batching and GPU instancing CSV headers.
const BENCHMARK_HEADERS = [
    // Test setup
    'testName',
    'mode',
    'objectCount',
    'repetition',

    // Performance result
    'avgFps',
    'avgFrameTimeMs',
    'runtimeDrawCallsAvg',
    'objectRenderMeshInstances',

    // Analyser result
    'analysedGroupCount',
    'analysedBatchingCandidateGroups',
    'analysedLargestGroupSize',

    // Optimisation result
    'optimisedAssignedObjectCount',
    'optimisedUsedGroupCount',
    'optimisedSkippedGroupCount'
];

export class BaselineBenchmarkRunner extends Script {
    static scriptName = 'baselineBenchmarkRunner';

    // Benchmark mode.
    // BASELINE = no optimisation.
    // BATCHING = applies batching after generation.
    // INSTANCING = applies GPU instancing after generation.
    /**
     * @attribute @type {E_BENCHMARK_MODE}
     */
    benchmarkMode = E_BENCHMARK_MODE.BASELINE;

    /**
     * @attribute @type {string}
     */
    testName = 'baseline_cube';

    // Object counts used for the benchmark.
    /**
     * @attribute @type {string}
     */
    objectCounts = '1000,2000,3000,4000,5000,6000,7000,8000,9000,10000';

    // Number of times each object count is tested.
    /**
     * @attribute @type {number}
     */
    repetitions = 3;

    // Warm-up time before measurement starts.
    /**
     * @attribute @type {number}
     */
    warmupSeconds = 3;

    // Measurement duration per run.
    /**
     * @attribute @type {number}
     */
    measureSeconds = 10;

    // Starts the benchmark automatically after the delay.
    /**
     * @attribute @type {boolean}
     */
    autoStart = false;

    // Delay before auto-starting the benchmark.
    /**
     * @attribute @type {number}
     */
    startDelaySeconds = 1;

    // Browser used for the test.
    /**
     * @attribute @type {string}
     */
    browser = 'Chrome';

    // Resolution used for the test.
    /**
     * @attribute @type {string}
     */
    resolution = '1920x1080';

    initialize() {
        this.generator = null;
        this.csvExporter = null;
        this.analyser = null;
        this.batchingOptimiser = null;
        this.instancingOptimiser = null;

        this.queue = [];
        this.currentRun = null;
        this.currentAnalysis = null;
        this.currentOptimisationResult = null;

        this.phase = 'idle';
        this.timer = 0;
        this.autoStartTimer = 0;
        this.hasAutoStarted = false;

        this.fpsSamples = [];
        this.frameTimeSamples = [];
        this.runtimeDrawCallSamples = [];

        console.log('[BASELINE BENCHMARK] Press B to start benchmark.');
    }

    postInitialize() {
        this.generator = this.entity.script.benchmarkSceneGenerator;
        this.csvExporter = this.entity.script.csvExporter;
        this.analyser = this.entity.script.objectAnalyser || null;
        this.batchingOptimiser = this.entity.script.batchingOptimiser || null;
        this.instancingOptimiser = this.entity.script.instancingOptimiser || null;
    }

    // Updates the benchmark flow.
    update(dt) {
        // waiting before auto-starting the benchmark.
        if (this.autoStart && !this.hasAutoStarted) {
            this.autoStartTimer += dt;

            if (this.autoStartTimer >= this.startDelaySeconds) {
                this.hasAutoStarted = true;
                this.startBenchmark();
            }
        }

        // starting the benchmark manually.
        if (this.app.keyboard && this.app.keyboard.wasPressed(KEY_B)) {
            this.startBenchmark();
        }

        if (this.phase === 'idle' || this.phase === 'done') {
            return;
        }

        // waiting during the warm-up phase.
        if (this.phase === 'warmup') {
            this.timer += dt;

            if (this.timer >= this.warmupSeconds) {
                this.startMeasurement();
            }

            return;
        }

        // collecting samples during the measurement phase.
        if (this.phase === 'measure') {
            this.timer += dt;
            this.collectSample(dt);

            if (this.timer >= this.measureSeconds) {
                this.finishMeasurement();
            }
        }
    }

    // Starts the full benchmark queue.
    startBenchmark() {
        if (!this.generator || !this.csvExporter) {
            console.error('[BASELINE BENCHMARK] Cannot start. Missing generator or CSV exporter.');
            return;
        }

        this.queue = this.buildQueue();
        this.csvExporter.startNew(BENCHMARK_HEADERS);

        console.log('[BASELINE BENCHMARK] Starting benchmark.');
        console.log('[BASELINE BENCHMARK] Test queue:', this.queue);

        this.startNextRun();
    }

    // Builds the benchmark queue.
    buildQueue() {
        const counts = this.objectCounts
            .split(',')
            .map(value => Number(value.trim()))
            .filter(value => Number.isFinite(value) && value > 0);

        const queue = [];

        // creating one run for every object count and repetition.
        for (const objectCount of counts) {
            for (let repetition = 1; repetition <= this.repetitions; repetition++) {
                queue.push({
                    objectCount,
                    repetition
                });
            }
        }

        return queue;
    }

    // Starts the next benchmark run.
    startNextRun() {
        if (!this.queue.length) {
            this.finishBenchmark();
            return;
        }

        this.currentRun = this.queue.shift();

        console.log(
            `[BASELINE BENCHMARK] Preparing ${this.currentRun.objectCount} objects, repetition ${this.currentRun.repetition}.`
        );

        this.generator.generate(this.currentRun.objectCount);

        // analysing the generated objects before applying optimisation.
        this.currentAnalysis = this.analyser ? this.analyser.analyse() : null;

        let optimisationResult = null;

        // applying batching when batching mode is selected.
        if (this.benchmarkMode === E_BENCHMARK_MODE.BATCHING) {
            if (!this.batchingOptimiser) {
                console.error('[BASELINE BENCHMARK] Cannot apply batching. Missing BatchingOptimiser.');
            } else {
                optimisationResult = this.batchingOptimiser.apply(this.currentAnalysis);
            }
        }

        // applying GPU instancing when instancing mode is selected.
        if (this.benchmarkMode === E_BENCHMARK_MODE.INSTANCING) {
            if (!this.instancingOptimiser) {
                console.error('[BASELINE BENCHMARK] Cannot apply instancing. Missing InstancingOptimiser.');
            } else {
                optimisationResult = this.instancingOptimiser.apply(this.currentAnalysis);
            }
        }

        this.currentOptimisationResult = optimisationResult;
        this.phase = 'warmup';
        this.timer = 0;

        this.fpsSamples.length = 0;
        this.frameTimeSamples.length = 0;
        this.runtimeDrawCallSamples.length = 0;
    }

    // Starts measuring the current run.
    startMeasurement() {
        console.log(
            `[BASELINE BENCHMARK] Measuring ${this.currentRun.objectCount} objects, repetition ${this.currentRun.repetition}.`
        );

        this.phase = 'measure';
        this.timer = 0;

        this.fpsSamples.length = 0;
        this.frameTimeSamples.length = 0;
        this.runtimeDrawCallSamples.length = 0;
    }

    // Collects one performance sample.
    collectSample(dt) {
        if (dt <= 0) {
            return;
        }

        const frameTimeMs = dt * 1000;
        const fps = 1 / dt;
        const runtimeDrawCalls = this.readRuntimeDrawCalls();

        this.fpsSamples.push(fps);
        this.frameTimeSamples.push(frameTimeMs);

        // storing draw calls only when PlayCanvas exposes them.
        if (runtimeDrawCalls !== null) {
            this.runtimeDrawCallSamples.push(runtimeDrawCalls);
        }
    }

    // Finishes the current measurement.
    finishMeasurement() {
        const fpsStats = this.calculateStats(this.fpsSamples);
        const frameTimeStats = this.calculateStats(this.frameTimeSamples);
        const drawCallStats = this.calculateStats(this.runtimeDrawCallSamples);
        const objectRenderMeshInstances = this.generator.countActiveMeshInstances();

        const analysis = this.currentAnalysis;
        const largestGroupSize = analysis && analysis.groups.length
            ? Math.max(...analysis.groups.map(group => group.count))
            : 0;

        // Baseline row.
        // const row = {
        //     testName: `${this.testName}_${this.currentRun.objectCount}`,
        //     mode: 'baseline',
        //     objectCount: this.currentRun.objectCount,
        //     meshVariantCount: 1,
        //     materialVariantCount: 1,
        //     repetition: this.currentRun.repetition,
        //     warmupSeconds: this.warmupSeconds,
        //     measureSeconds: this.measureSeconds,
        //     avgFps: this.formatNumber(fpsStats.avg),
        //     minFps: this.formatNumber(fpsStats.min),
        //     maxFps: this.formatNumber(fpsStats.max),
        //     avgFrameTimeMs: this.formatNumber(frameTimeStats.avg),
        //     minFrameTimeMs: this.formatNumber(frameTimeStats.min),
        //     maxFrameTimeMs: this.formatNumber(frameTimeStats.max),
        //     runtimeDrawCallsAvg: drawCallStats.count > 0 ? this.formatNumber(drawCallStats.avg) : 'n/a',
        //     runtimeDrawCallsMin: drawCallStats.count > 0 ? this.formatNumber(drawCallStats.min) : 'n/a',
        //     runtimeDrawCallsMax: drawCallStats.count > 0 ? this.formatNumber(drawCallStats.max) : 'n/a',
        //     objectRenderMeshInstances: objectRenderMeshInstances,
        //     browser: this.browser,
        //     resolution: this.resolution,
        //     deviceNotes: this.deviceNotes,
        //     notes: drawCallStats.count > 0
        //         ? 'Runtime draw calls read from app stats.'
        //         : 'Runtime draw calls unavailable from app stats; use MiniStats screenshot and objectRenderMeshInstances.'
        // };

        // Analyser-only row.
        // const row = {
        //     testName: `${this.testName}_${this.currentRun.objectCount}`,
        //     mode: 'analyser',
        //     objectCount: this.currentRun.objectCount,
        //     repetition: this.currentRun.repetition,
        //     analysedObjectCount: analysis ? analysis.objectCount : 'n/a',
        //     analysedRenderComponentCount: analysis ? analysis.renderComponentCount : 'n/a',
        //     analysedMeshInstanceCount: analysis ? analysis.meshInstanceCount : 'n/a',
        //     analysedGroupCount: analysis ? analysis.groupCount : 'n/a',
        //     analysedInstancingCandidateGroups: analysis ? analysis.instancingCandidateCount : 'n/a',
        //     analysedBatchingCandidateGroups: analysis ? analysis.batchingCandidateCount : 'n/a',
        //     analysedLargestGroupSize: analysis ? largestGroupSize : 'n/a'
        // };

        // Batching and GPU instancing row.
        const row = {
            // Test setup
            testName: `${this.testName}_${this.currentRun.objectCount}`,
            mode: this.getModeName(),
            objectCount: this.currentRun.objectCount,
            repetition: this.currentRun.repetition,

            // Performance result
            avgFps: this.formatNumber(fpsStats.avg),
            avgFrameTimeMs: this.formatNumber(frameTimeStats.avg),
            runtimeDrawCallsAvg: drawCallStats.count > 0 ? this.formatNumber(drawCallStats.avg) : 'n/a',
            objectRenderMeshInstances: objectRenderMeshInstances,

            // Analyser result
            analysedGroupCount: analysis ? analysis.groupCount : 'n/a',
            analysedBatchingCandidateGroups: analysis ? analysis.batchingCandidateCount : 'n/a',
            analysedLargestGroupSize: analysis ? largestGroupSize : 'n/a',

            // Optimisation result
            optimisedAssignedObjectCount: this.currentOptimisationResult
                ? this.currentOptimisationResult.assignedObjectCount
                : 0,

            optimisedUsedGroupCount: this.currentOptimisationResult
                ? this.currentOptimisationResult.usedGroupCount
                : 0,

            optimisedSkippedGroupCount: this.currentOptimisationResult
                ? this.currentOptimisationResult.skippedGroupCount
                : 0
        };

        this.csvExporter.addRow(row);

        console.log('[BASELINE BENCHMARK] Finished run:', row);

        this.phase = 'between';
        this.timer = 0;

        // giving the browser one moment between tests.
        setTimeout(() => {
            this.startNextRun();
        }, 250);
    }

    // Finishes the full benchmark.
    finishBenchmark() {
        this.phase = 'done';

        const csv = this.csvExporter.finish();

        console.log('[BASELINE BENCHMARK] Benchmark finished.');
        console.log('[BASELINE BENCHMARK] Copy/download the CSV and save it in Raw results/.');

        return csv;
    }

    // Calculates basic stats from samples.
    calculateStats(samples) {
        if (!samples || !samples.length) {
            return {
                count: 0,
                avg: 0,
                min: 0,
                max: 0
            };
        }

        let sum = 0;
        let min = Number.POSITIVE_INFINITY;
        let max = Number.NEGATIVE_INFINITY;

        // calculating sum, minimum, and maximum
        for (const value of samples) {
            sum += value;
            min = Math.min(min, value);
            max = Math.max(max, value);
        }

        return {
            count: samples.length,
            avg: sum / samples.length,
            min,
            max
        };
    }

    // Reads runtime draw calls from PlayCanvas stats.
    readRuntimeDrawCalls() {
        const stats = this.app.stats;

        if (!stats) {
            return null;
        }

        const candidates = [
            stats.drawCalls,
            stats.render?.drawCalls,
            stats.render?.calls,
            stats.frame?.drawCalls,
            stats.graphics?.drawCalls,
            stats.scene?.drawCalls
        ];

        // checking different possible stats paths because this can vary.
        for (const candidate of candidates) {
            const value = this.extractNumber(candidate);

            if (value !== null) {
                return value;
            }
        }

        return null;
    }

    // Extracts a number from a stats value.
    extractNumber(value) {
        if (typeof value === 'number' && Number.isFinite(value)) {
            return value;
        }

        if (!value || typeof value !== 'object') {
            return null;
        }

        const possibleKeys = ['total', 'count', 'value', 'drawCalls', 'forward'];

        // checking common keys used by stats objects.
        for (const key of possibleKeys) {
            if (typeof value[key] === 'number' && Number.isFinite(value[key])) {
                return value[key];
            }
        }

        return null;
    }

    // Formats a number for CSV output.
    formatNumber(value) {
        if (!Number.isFinite(value)) {
            return 'n/a';
        }

        return Number(value.toFixed(3));
    }

    // Gets the current benchmark mode name.
    getModeName() {
        if (this.benchmarkMode === E_BENCHMARK_MODE.BATCHING) {
            return 'batching';
        }

        if (this.benchmarkMode === E_BENCHMARK_MODE.INSTANCING) {
            return 'instancing';
        }

        return 'baseline';
    }
}
